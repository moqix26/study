## §11 完整猜数游戏

```cpp
#include <iostream>
#include <cstdlib>
#include <ctime>
int main() {
    std::srand(static_cast<unsigned>(std::time(nullptr)));
    int secret = std::rand() % 100 + 1;
    int guess = 0, tries = 0;
    std::cout << "猜 1～100\n";
    while (guess != secret) {
        std::cout << "输入: ";
        if (!(std::cin >> guess)) {
            std::cin.clear();
            std::cin.ignore(10000, '\n');
            continue;
        }
        ++tries;
        if (guess < secret) std::cout << "太小\n";
        else if (guess > secret) std::cout << "太大\n";
    }
    std::cout << "对了！" << tries << " 次\n";
    return 0;
}
```

**精读**：srand 播种；rand%100+1；cin 失败 clear+ignore；continue 重试。
